#+begin_src C -n
  DEFUN ("bolp", Fbolp, Sbolp, 0, 0, 0,
         doc: /* Return t if point is at the beginning of a line.  */)
      (void)
  {
      if (PT == BEGV || FETCH_BYTE (PT_BYTE - 1) == '\n')
          return Qt;
      return Qnil;
  }
#+end_src
